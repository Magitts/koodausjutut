	käyttö estetty:
	jaa kansio jossa verkkosivun tiedot ovat
	kopioi web.config verkkosivu kansioon

	nyt pitäisi tulla ilmoitus 403.14 forbidden, joka tarkoittaa tässä että kansioita ei saa selata
		muokkaa web.config
		lisää system.webServer tagien väliin
<system.webServer>
    <directoryBrowse enabled="true" />
</system.webServer>
		ongelman pitäisi kuitenkin korjaantua jos lisäät jonkin verkkosivun verkkosivu kansioon.

muokkaamattomana valmiin config tiedoston pitäisi näyttää tältä

<?xml version="1.0" encoding="UTF-8"?>
<configuration>
    <system.webServer>
        <handlers accessPolicy="Read, Execute, Script">
            <add name="python" path=".py" verb="*" modules="CgiModule" scriptProcessor="C:\Program Files\Python36\python.exe" resourceType="File" />
        </handlers>
    </system.webServer>
</configuration>
